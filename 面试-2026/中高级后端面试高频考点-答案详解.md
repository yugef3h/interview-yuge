## Q1：消息不丢失怎么保证？（全链路三阶段）

**口诀：生产刷盘消费ACK，全链路一个都不能少**

```python
"""
全链路防丢失 = 生产端确认 + Broker刷盘 + 消费端手动ACK
"""
import pika  # RabbitMQ Python 客户端

# ========== 生产端：同步发送 + 确认回调 ==========
def producer_reliable():
    connection = pika.BlockingConnection(pika.ConnectionParameters('localhost'))
    channel = connection.channel()

    # 开启发布确认模式（类似 RocketMQ 的同步发送）
    channel.confirm_delivery()

    # 队列持久化
    channel.queue_declare(queue='order_queue', durable=True)

    try:
        # delivery_mode=2 表示消息持久化到磁盘
        channel.basic_publish(
            exchange='',
            routing_key='order_queue',
            body='{"order_id": 123}',
            properties=pika.BasicProperties(
                delivery_mode=2,  # 消息持久化
            )
        )
        print("发送成功，Broker已确认")
    except Exception as e:
        print(f"发送失败，重试: {e}")
        # 重试逻辑：指数退避，最多3次
        # ...

# ========== 消费端：手动ACK（处理完再确认）==========
def consumer_reliable():
    connection = pika.BlockingConnection(pika.ConnectionParameters('localhost'))
    channel = connection.channel()

    def callback(ch, method, properties, body):
        try:
            # 1. 处理业务逻辑
            process_order(body)
            # 2. 处理成功 → 手动ACK（关键！不要 auto_ack=True）
            ch.basic_ack(delivery_tag=method.delivery_tag)
            print("消费成功，已ACK")
        except Exception:
            # 3. 处理失败 → NACK，消息重新入队 或 进入死信队列
            ch.basic_nack(delivery_tag=method.delivery_tag, requeue=False)
            print("消费失败，进入死信队列")

    # auto_ack=False 是关键：手动确认
    channel.basic_consume(queue='order_queue', on_message_callback=callback, auto_ack=False)
    channel.start_consuming()
```

**记忆公式：**

```
生产端（confirm）→ Broker（sync_flush + 多副本）→ 消费端（手动ACK）
     ↓                    ↓                              ↓
  失败重试           刷盘+ISR≥2                    失败进死信
```

## Q2：消息重复消费怎么解决？（幂等）

**口诀：消息必然重复，消费必须幂等。唯一索引最可靠，SETNX最简单。**

```python
"""
幂等方案：同样消息消费N次，结果和消费1次一样
"""
import redis
import sqlite3

r = redis.Redis()

# ====== 方案1：数据库唯一索引（最可靠）======
def idempotent_by_db(msg_id: str, biz_type: str, data: dict):
    """msg_id + biz_type 联合唯一键，插入即幂等"""
    conn = sqlite3.connect('business.db')
    try:
        conn.execute(
            "INSERT INTO order_event (msg_id, biz_type, data) VALUES (?, ?, ?)",
            (msg_id, biz_type, str(data))
        )
        # 插入成功 → 第一次消费，执行业务逻辑
        conn.commit()
        do_business_logic(data)
    except sqlite3.IntegrityError:
        # 唯一键冲突 → 重复消费，跳过
        print(f"消息 {msg_id} 已处理过，跳过")
    finally:
        conn.close()


# ====== 方案2：Redis SETNX（最常用）======
def idempotent_by_redis(msg_id: str, data: dict):
    """SETNX 原子判断：key不存在才设置，存在就返回False"""
    key = f"msg_consumed:{msg_id}"
    # NX=True: 只有key不存在时才设置
    # EX=3600: 过期时间1小时（防止key无限增长）
    if r.set(key, "1", nx=True, ex=3600):
        do_business_logic(data)  # 第一次，执行
    else:
        print("重复消费，跳过")  # 已经消费过


# ====== 方案3：版本号（乐观锁思想）======
def idempotent_by_version(order_id: int, new_status: str, version: int):
    """只有版本号匹配才更新，防止旧消息覆盖新状态"""
    conn = sqlite3.connect('business.db')
    affected = conn.execute(
        "UPDATE orders SET status=?, version=version+1 "
        "WHERE id=? AND version < ?",
        (new_status, order_id, version)
    ).rowcount
    conn.commit()
    if affected == 0:
        print("版本号过期，跳过（说明已处理过更新的消息）")
```

**一句话总结：** 重复消费是 MQ 的 "At-Least-Once" 特性，不是 bug。用唯一索引 / SETNX / 版本号 / 状态机保证幂等。

## Q4：RocketMQ 事务消息原理？

**口诀：半消息先占位，本地事务决定提交或回滚，超时回查补一刀。**

```python
"""
RocketMQ 事务消息 = 两阶段提交
阶段一：发半消息（Half Message，消费者不可见）
阶段二：执行本地事务 → 成功Commit / 失败Rollback / 未知等回查
"""

class TransactionListener:
    """
    RocketMQ 事务消息监听器（Python 伪代码，展示核心逻辑）
    实际使用 rocketmq-client-python 库
    """

    def execute_local_transaction(self, msg):
        """执行本地事务"""
        try:
            # 本地数据库操作：创建订单
            order_id = create_order_in_db(msg)
            # 成功 → Commit，消息对消费者可见
            return "COMMIT"
        except Exception:
            # 失败 → Rollback，消息删除
            return "ROLLBACK"
        # 如果返回 UNKNOWN，Broker 会定时回查

    def check_local_transaction(self, msg):
        """Broker 回查（本地事务超时/返回UNKNOWN时触发）"""
        # 查数据库：订单到底创建成功没有？
        order = query_order_from_db(msg['order_id'])
        if order and order['status'] == 'created':
            return "COMMIT"   # 查到了 → 提交
        else:
            return "ROLLBACK" # 没查到 → 回滚


# 关键数字记忆：
# - 回查最多 15 次
# - 如果 15 次都无响应 → 直接标记 ROLLBACK
```

**流程图记法：**

```
Producer → Broker（半消息，消费者不可见）
              ↓
         执行本地事务
         ╱    │    ╲
      成功    失败   超时/未知
       ↓       ↓       ↓
    COMMIT  ROLLBACK  回查(×15次)
       ↓       ↓       ↓
   消费者可见  删除   最终决定 COMMIT/ROLLBACK
```

## Q9：延迟任务怎么实现？

**口诀：RocketMQ 18等级（最方便），Redis ZSET（最灵活），时间轮（最高效），DB轮询（最简单）。**

```python
"""
5 种延迟任务方案 Python 实现
"""
import time
import redis
import threading
import heapq

r = redis.Redis()

# ====== 方案1：Redis ZSET（最灵活，精度高）======
def add_delay_task(task_id: str, execute_at: float):
    """score = 执行时间戳，加入有序集合"""
    r.zadd("delay_queue", {task_id: execute_at})

def poll_delay_task():
    """定时轮询，弹出到期任务"""
    now = time.time()
    # ZRANGEBYSCORE：取出 score <= now 的所有任务
    tasks = r.zrangebyscore("delay_queue", 0, now)
    for task_id in tasks:
        # ZREM 原子删除（防止并发重复取）
        if r.zrem("delay_queue", task_id):
            execute_task(task_id)


# ====== 方案2：时间轮（TimeWheel，最高效）======
class TimeWheel:
    """时间轮 = 数组 + 链表，Netty HashedWheelTimer 的简化版"""
    def __init__(self, tick_ms: int = 100, wheel_size: int = 60):
        self.tick_ms = tick_ms          # 每格 100ms
        self.wheel_size = wheel_size    # 60 格 = 6 秒一圈
        self.slots = [[] for _ in range(wheel_size)]
        self.current_slot = 0

    def add_task(self, delay_ms: int, task):
        """计算延迟几格，放入对应槽位"""
        ticks = delay_ms // self.tick_ms
        slot = (self.current_slot + ticks) % self.wheel_size
        self.slots[slot].append(task)

    def tick(self):
        """每个 tick 推进一格，执行当前槽位所有任务"""
        for task in self.slots[self.current_slot]:
            execute(task)
        self.slots[self.current_slot] = []  # 清空
        self.current_slot = (self.current_slot + 1) % self.wheel_size


# ====== 方案3：数据库轮询（最简单但性能差）======
def db_poll_delay_task():
    """轮询数据库，查到期任务"""
    conn = sqlite3.connect('business.db')
    tasks = conn.execute(
        "SELECT id, payload FROM delay_tasks "
        "WHERE execute_time <= datetime('now') AND status='pending' "
        "LIMIT 100"
    ).fetchall()
    for task in tasks:
        # 乐观锁抢任务
        affected = conn.execute(
            "UPDATE delay_tasks SET status='processing' "
            "WHERE id=? AND status='pending'", (task[0],)
        ).rowcount
        if affected:
            execute(task[1])
```

**延迟方案选择：**
- 18个固定等级 → RocketMQ 延迟等级
- 任意时间精度 → Redis ZSET
- 海量短延迟 → 时间轮
- 简单低频 → DB 轮询

# 二、分布式基础理论

## 2.1 CAP 定理

**口诀：P必须选（网络必挂），实际是CP vs AP的选择。ZK是CP，Eureka是AP。**

```python
"""
CAP 定理 = Consistency(一致性) + Availability(可用性) + Partition Tolerance(分区容错)
三者只能同时满足两个

P（分区容错）是必须选的：网络故障不可避免！

所以实际只有两种选择：
- CP：发生网络分区时，宁可不可用，也要保证数据一致 → ZK、etcd
- AP：发生网络分区时，宁可数据不一致，也要保证服务可用 → Eureka
"""

                return self.get_services_from_cache()  # 本地缓存兜底
```

## 2.3 Raft 一致性算法

**口诀：Leader干活+Follower抄作业，过半确认就提交。选举靠超时+投票。**

```python
"""
Raft = 选主 + 日志复制
- 角色：Leader（干活）、Follower（抄作业）、Candidate（竞选者）
- 选举：超时没心跳 → 变成Candidate → 拉票 → 过半数就当Leader
- 日志：写Leader → 复制给Follower → 过半确认 → 提交
"""
import random
import time
import threading

class RaftNode:
    """Raft 节点 Python 实现（简化版，演示原理）"""

    def __init__(self, node_id: int):
        self.node_id = node_id
        self.role = "follower"       # leader / follower / candidate
        self.current_term = 0        # 当前任期
        self.voted_for = None        # 这轮投给谁了
        self.log = []                # 本地日志
        self.commit_index = -1       # 已提交的位置
        self.last_applied = -1       # 已应用到状态机的位置

        # 选举超时：150~300ms 随机（避免同时发起选举形成平票）
        self.election_timeout = random.uniform(0.15, 0.30)

    def become_candidate(self):
        """超时未收到心跳 → 发起选举"""
        self.role = "candidate"
        self.current_term += 1       # 任期 +1
        self.voted_for = self.node_id  # 先投自己
        votes = 1  # 自己一票
        # 向其他节点索要投票...
        return votes

    def become_leader(self, votes: int, total_nodes: int):
        """收到多数票 → 成为 Leader"""
        if votes > total_nodes / 2:  # 过半！
            self.role = "leader"
            # 开始发送心跳，阻止其他节点超时选举
            self.start_heartbeat()

    def append_entries(self, entries):
        """Leader 复制日志给 Follower"""
        # 1. Leader 追加日志到本地
        self.log.extend(entries)
        # 2. 并行发送 AppendEntries 给所有 Follower
        # 3. 过半 Follower 确认 → 提交
        # 4. 提交后才返回 Client 成功
        pass

    def start_heartbeat(self):
        """Leader 定时发心跳（空的 AppendEntries）"""
        def heartbeat_loop():
            while self.role == "leader":
                # 广播心跳给所有 Follower
                time.sleep(0.05)  # 50ms 一次
        threading.Thread(target=heartbeat_loop, daemon=True).start()


# ==== 面试核心数字 ====
# 选举超时：150~300ms 随机
# 心跳间隔：50ms（远小于选举超时）
# 提交条件：过半节点确认（quorum = N/2 + 1）
# 保证：顺序一致性（不是线性一致性）
```

**面试必答：Raft 保证的是顺序一致性（所有节点按相同顺序执行命令），不是线性一致性（读可能读到旧值）。**

## 2.4 分布式ID生成

**口诀：雪花算法是主流，1+41+10+12=64位。时钟回拨三种解决：等、换、借。**

```python
"""
分布式ID生成方案 Python 实现
"""
import time
import uuid
import threading

# ====== 方案1：UUID（简单但数据库不友好）======
def generate_uuid():
    """UUID 非单调递增，B+Tree 插入性能差（页分裂）"""
    return str(uuid.uuid4())

# ====== 方案2：雪花算法（Snowflake，最主流）======
class Snowflake:
    """
    结构（64位）：
    1 bit 不用 + 41 bit 时间戳 + 10 bit 机器ID + 12 bit 序列号
    ┌─┬─────────────────────┬────────────┬──────────────┐
    │0│  41位毫秒时间戳      │ 10位机器ID  │ 12位序列号    │
    └─┴─────────────────────┴────────────┴──────────────┘

    单机 QPS：4096/ms = 约 400万/s
    """
    def __init__(self, worker_id: int, datacenter_id: int):
        self.worker_id = worker_id          # 5 bit，最多32台机器
        self.datacenter_id = datacenter_id  # 5 bit，最多32个数据中心
        self.sequence = 0                   # 12 bit，每毫秒4096个
        self.last_timestamp = -1

        # 起始时间戳（2020-01-01 00:00:00）
        self.twepoch = 1577808000000

    def _current_millis(self) -> int:
        return int(time.time() * 1000)

    def next_id(self) -> int:
        timestamp = self._current_millis()

        # ==== 时钟回拨处理 ====
        if timestamp < self.last_timestamp:
            offset = self.last_timestamp - timestamp
            if offset <= 5:
                # 策略1：短期等待（回拨 < 5ms，等一等）
                time.sleep(offset / 1000.0)
                timestamp = self._current_millis()
            else:
                # 策略2：扩展位借用（借序列号的高位凑）
                # 策略3：备份节点替换（换个机器ID）
                raise ClockBackError(f"时钟回拨{offset}ms，等待中...")

        if timestamp == self.last_timestamp:
            self.sequence = (self.sequence + 1) & 0xFFF  # 4095 内循环
            if self.sequence == 0:
                # 序列号用完了，等到下一毫秒
                while timestamp <= self.last_timestamp:
                    timestamp = self._current_millis()
        else:
            self.sequence = 0

        self.last_timestamp = timestamp

        # 组装 ID
        return (
            ((timestamp - self.twepoch) << 22) |
            (self.datacenter_id << 17) |
            (self.worker_id << 12) |
            self.sequence
        )


# ====== 方案3：号段模式（Leaf）======
class LeafSegment:
    """
    号段模式：一次从DB取一段ID（如1000个），用完了再取
    优点：高QPS、趋势递增
    缺点：依赖DB
    """
    def __init__(self):
        self.current_id = 0
        self.max_id = 0

    def next_id(self):
        if self.current_id >= self.max_id:
            self._fetch_segment()  # 用完了，去DB取下一段
        self.current_id += 1
        return self.current_id

    def _fetch_segment(self):
        # UPDATE leaf SET max_id = max_id + 1000 WHERE biz='order'
        # 原子操作，多实例不冲突
        self.current_id = self.max_id  # 从上次结束的位置开始
        self.max_id += 1000
```

**记忆公式：**

```
雪花算法 64bit = 1(不用) + 41(时间戳|69年) + 10(机器ID|1024台) + 12(序列号|4096/ms)
时钟回拨：等等(≤5ms) → 借位 → 换机器
```

```python
"""
服务注册发现 = 注册（Provider启动时注册）+ 发现（Consumer订阅+缓存）
"""
import time
import threading
import requests


class ServiceRegistry:
    """简化的注册中心（Nacos/Consul 原理演示）"""

    def __init__(self):
        self.services = {}  # service_name → [{"ip":..., "port":..., "last_heartbeat":...}]

    def register(self, service_name: str, ip: str, port: int):
        """Provider 启动时注册"""
        instance = {"ip": ip, "port": port, "last_heartbeat": time.time()}
        if service_name not in self.services:
            self.services[service_name] = []
        self.services[service_name].append(instance)
        print(f"[注册] {service_name} @ {ip}:{port}")

    def heartbeat(self, service_name: str, ip: str, port: int):
        """Provider 定时心跳（每5秒）"""
        for inst in self.services.get(service_name, []):
            if inst["ip"] == ip and inst["port"] == port:
                inst["last_heartbeat"] = time.time()

    def discover(self, service_name: str) -> list:
        """Consumer 发现服务"""
        self._cleanup_dead(service_name)  # 剔除心跳超时（30s）的节点
        return self.services.get(service_name, [])

    def _cleanup_dead(self, service_name: str):
        """清理死节点"""
        now = time.time()
        self.services[service_name] = [
            inst for inst in self.services.get(service_name, [])
            if now - inst["last_heartbeat"] < 30  # 30s 无心跳 = 死了
        ]
```

### 熔断（Circuit Breaker）

**口诀：三状态循环→关→开→半开→关。失败多就开，时间到就试探。**

```python
"""
熔断器 = 自动挡保险丝
CLOSED(关:正常) → OPEN(开:直接拒绝) → HALF_OPEN(半开:试探) → CLOSED
"""
import time
import functools

class CircuitBreaker:
    """熔断器 Python 实现"""

    def __init__(self, failure_threshold: int = 5, cooldown_seconds: float = 10.0):
        self.failure_threshold = failure_threshold  # 失败5次就熔断
        self.cooldown_seconds = cooldown_seconds    # 冷却10秒
        self.failure_count = 0
        self.last_failure_time = 0
        self.state = "CLOSED"  # CLOSED → OPEN → HALF_OPEN

    def call(self, func, *args, **kwargs):
        """包装函数调用，自动熔断"""
        if self.state == "OPEN":
            if time.time() - self.last_failure_time > self.cooldown_seconds:
                self.state = "HALF_OPEN"
                print("[熔断] OPEN → HALF_OPEN，放少量请求试探")
            else:
                raise Exception("熔断中，直接拒绝")

        try:
            result = func(*args, **kwargs)
            # 成功！
            if self.state == "HALF_OPEN":
                self.state = "CLOSED"
                self.failure_count = 0
                print("[熔断] HALF_OPEN → CLOSED，恢复正常")
            return result
        except Exception as e:
            self.failure_count += 1
            self.last_failure_time = time.time()

            if self.state == "CLOSED" and self.failure_count >= self.failure_threshold:
                self.state = "OPEN"
                print(f"[熔断] CLOSED → OPEN，连续失败{self.failure_count}次")
            elif self.state == "HALF_OPEN":
                self.state = "OPEN"  # 试探失败，回到熔断
                print("[熔断] HALF_OPEN → OPEN，试探失败")

            raise e
```

### 限流算法

**口诀：令牌桶允突发，漏桶强制平滑，滑动窗口更精确。**

```python
"""
三种限流算法 Python 实现
"""
import time
from collections import deque

# ====== 令牌桶（Token Bucket）======
# 匀速生成令牌，请求取令牌。允许突发（桶里有积攒的令牌）
class TokenBucket:
    def __init__(self, rate: int, capacity: int):
        self.rate = rate          # 每秒生成 rate 个令牌
        self.capacity = capacity  # 桶最多装 capacity 个令牌
        self.tokens = capacity    # 当前令牌数（初始装满）
        self.last_time = time.time()

    def allow(self) -> bool:
        """来一个请求，试试能不能拿走一个令牌"""
        now = time.time()
        # 按时间流逝生成新令牌
        elapsed = now - self.last_time
        self.tokens = min(self.capacity, self.tokens + elapsed * self.rate)
        self.last_time = now

        if self.tokens >= 1:
            self.tokens -= 1  # 拿走一个令牌
            return True        # 放行
        return False           # 限流

# 使用：100 QPS，允许突发到200
limiter = TokenBucket(rate=100, capacity=200)


# ====== 漏桶（Leaky Bucket）======
# 请求进桶，匀速流出。绝对平滑，不允许突发
class LeakyBucket:
    def __init__(self, rate: int, capacity: int):
        self.rate = rate          # 每秒流出 rate 个
        self.capacity = capacity  # 桶容量
        self.water = 0            # 当前水位
        self.last_leak = time.time()

    def allow(self) -> bool:
        now = time.time()
        # 先漏水
        elapsed = now - self.last_leak
        self.water = max(0, self.water - elapsed * self.rate)
        self.last_leak = now

        if self.water < self.capacity:
            self.water += 1  # 进水
            return True
        return False


# ====== 滑动窗口（Sliding Window）======
# 统计过去 N 秒的请求数，精准限流
class SlidingWindow:
    def __init__(self, window_seconds: int, max_requests: int):
        self.window_seconds = window_seconds
        self.max_requests = max_requests
        self.timestamps = deque()

    def allow(self) -> bool:
        now = time.time()
        cutoff = now - self.window_seconds
        # 清除超时的记录
        while self.timestamps and self.timestamps[0] < cutoff:
            self.timestamps.popleft()

        if len(self.timestamps) < self.max_requests:
            self.timestamps.append(now)
            return True
        return False

# 对比记忆：
# 令牌桶：适合允许突发的场景（API网关）
# 漏桶：适合绝对平滑的场景（流量整形）
# 滑动窗口：适合精确控制的场景（防刷）
```

## 3.4 全链路追踪

**口诀：TraceID串全场，SpanID标单次，ParentSpanID画父子。透传靠Header/Metadata。**

```python
"""
全链路追踪 = TraceID（全局ID）+ SpanID（一次调用）+ ParentSpanID（父调用）
一次请求的所有 Span 串起来就是调用链
"""
import uuid
import threading
import json

# ====== TraceID 透传 ======
class TraceContext:
    """追踪上下文，跟随请求跨服务/跨线程/跨MQ"""

    _local = threading.local()  # 线程隔离

    @classmethod
    def set(cls, trace_id: str, span_id: str):
        cls._local.trace_id = trace_id
        cls._local.span_id = span_id

    @classmethod
    def get_trace_id(cls) -> str:
        return getattr(cls._local, 'trace_id', str(uuid.uuid4()))

    @classmethod
    def get_span_id(cls) -> str:
        return getattr(cls._local, 'span_id', str(uuid.uuid4()))


def http_middleware(request, next_handler):
    """HTTP 透传：从 Header 提取 TraceID"""
    trace_id = request.headers.get("X-Trace-Id", str(uuid.uuid4()))
    span_id = str(uuid.uuid4())[:8]
    TraceContext.set(trace_id, span_id)

    # 调用下游时透传
    response = requests.get(
        "http://downstream-service/api/data",
        headers={"X-Trace-Id": trace_id}  # ← 透传！
    )
    return next_handler(request)


def mq_producer(message: dict):
    """MQ 透传：放在消息头"""
    msg = {
        "body": message,
        "header": {
            "trace_id": TraceContext.get_trace_id(),  # ← 透传！
            "span_id": TraceContext.get_span_id()
        }
    }
    rabbitmq_channel.basic_publish(
        exchange='',
        routing_key='order_queue',
        body=json.dumps(msg)
    )


def thread_propagate():
    """线程池透传：父线程 → 子线程"""
    # 方式1：显式传递
    trace_id = TraceContext.get_trace_id()
    def worker(tid):
        TraceContext.set(tid, str(uuid.uuid4())[:8])
        do_work()
    threading.Thread(target=worker, args=(trace_id,)).start()

    # 方式2：logging MDC / contextvars（Python 3.7+）
    import contextvars
```

```python
# V3：看门狗自动续期（Redisson 原理）
class RedissonLikeLock:
    """带看门狗的 Redis 锁"""

    def __init__(self, redis_client, key: str, ttl: int = 30):
        self.redis = redis_client
        self.key = key
        self.value = str(uuid.uuid4())
        self.ttl = ttl
        self.watchdog_running = False

    def acquire(self) -> bool:
        if self.redis.set(self.key, self.value, nx=True, ex=self.ttl):
            self._start_watchdog()  # 启动看门狗！
            return True
        return False

    def _start_watchdog(self):
        """看门狗线程：每 ttl/3 秒续期一次"""
        def renew():
            self.watchdog_running = True
            while self.watchdog_running:
                time.sleep(self.ttl / 3)  # 每10秒
                # Lua 原子续期：只有 value 匹配才续期
                lua = """
                if redis.call('get', KEYS[1]) == ARGV[1] then
                    return redis.call('expire', KEYS[1], ARGV[2])
                else
                    return 0
                end
                """
                self.redis.eval(lua, 1, self.key, self.value, self.ttl)

        threading.Thread(target=renew, daemon=True).start()

    def release(self):
        self.watchdog_running = False  # 停止续期
        lua = """
        if redis.call('get', KEYS[1]) == ARGV[1] then
            return redis.call('del', KEYS[1])
        else
            return 0
        end
        """
        self.redis.eval(lua, 1, self.key, self.value)

# 优点：公平锁、断连自动释放（临时节点）
```

## 5.1 缓存三问题

**口诀：穿透查不到用布隆，击穿热点用互斥锁，雪崩大批过期加随机。**

```python
"""
缓存三问题 = 穿透（查不存在）+ 击穿（热点过期）+ 雪崩（大批过期）
"""
import redis
import hashlib
import random
import threading

r = redis.Redis()

# ====== 穿透：查不存在的数据 → 请求直达DB ======
# 场景：黑客一直查 id=-1 的用户，缓存没命中 → 每次都查DB
class BloomFilter:
    """布隆过滤器：判断 key 绝对不在集合中"""
    def __init__(self, size: int = 1000000, hash_count: int = 3):
        self.bit_array = [0] * size
        self.size = size
        self.hash_count = hash_count

    def _hashes(self, key: str):
        for i in range(self.hash_count):
            h = hashlib.md5(f"{key}_{i}".encode()).hexdigest()
            yield int(h, 16) % self.size

    def add(self, key: str):
        for pos in self._hashes(key):
            self.bit_array[pos] = 1

    def might_contain(self, key: str) -> bool:
        """返回 False = 绝对不存在；返回 True = 可能存在（有小概率误判）"""
        return all(self.bit_array[pos] == 1 for pos in self._hashes(key))


bloom = BloomFilter()
# 初始化：把所有存在的 ID 加入布隆过滤器
for user_id in get_all_user_ids():
    bloom.add(str(user_id))

def query_user_safe(user_id: int):
    """防穿透查询"""
    # 步骤1：布隆过滤器判断
    if not bloom.might_contain(str(user_id)):
        return None  # 绝对不存在，不查DB！

    # 步骤2：查缓存
    cached = r.get(f"user:{user_id}")
    if cached:
        return cached

    # 步骤3：查DB
    user = db_query(user_id)
    if user:
        r.set(f"user:{user_id}", str(user), ex=3600)
    else:
        # 缓存空值（短TTL，防止缓存穿透）
        r.set(f"user:{user_id}", "NULL", ex=60)
    return user


# ====== 击穿：热点Key过期 → 大量请求直达DB ======
# 场景：热门商品缓存刚好过期，瞬间1000个请求查DB
hot_product_lock = threading.Lock()

def query_hot_product(product_id: int):
    """防击穿：互斥锁，一个线程去查DB，其他等结果"""
    cached = r.get(f"product:{product_id}")
    if cached:
        return cached

    # 只有一个线程去查DB
    lock_key = f"lock:product:{product_id}"
    if r.set(lock_key, "1", nx=True, ex=10):  # 抢到锁
        try:
            product = db_query(product_id)      # 只有我去查DB
            r.set(f"product:{product_id}", str(product), ex=3600)  # 写缓存
            return product
        finally:
            r.delete(lock_key)
    else:
        # 没抢到锁 → 等一会再查缓存
        time.sleep(0.05)
        return r.get(f"product:{product_id}")


# ====== 雪崩：大量Key同时过期 → DB瞬间压垮 ======
def set_with_random_ttl(key: str, value: str, base_ttl: int = 3600):
    """防雪崩：过期时间 = 基础值 + 随机值"""
    random_offset = random.randint(0, 600)  # 0~10分钟随机
    r.set(key, value, ex=base_ttl + random_offset)
```

**一图记三问题：**

```
穿透 → 查不存在 → 布隆过滤器 + 缓存空值
击穿 → 热点过期 → 互斥锁（单线程查DB）
雪崩 → 大批过期 → 过期时间加随机值
```

## 6.1 分库分表

**口诀：500万行/2GB考虑分表，2000QPS考虑分库。垂直按业务拆，水平按片键拆。**

```python
"""
分库分表 Python 演示
"""
import hashlib

# ====== 分片算法 ======

def shard_by_mod(key: int, shard_count: int) -> int:
    """取模分片：key % N
    优点：简单均匀
    缺点：扩容需要迁移全部数据（N变了）
    """
    return key % shard_count

def shard_by_hash(key: str, shard_count: int) -> int:
    """哈希取模：hash(key) % N
    优点：分布均匀
    缺点：扩容需迁移数据
    """
    h = hashlib.md5(key.encode()).hexdigest()
    return int(h, 16) % shard_count

def shard_by_range(key: int, ranges: list) -> int:
    """范围分片：按ID范围
    优点：扩容方便（新数据到新分片）
    缺点：热点（新数据集中在最新分片）
    """
    for i, (lo, hi) in enumerate(ranges):
        if lo <= key < hi:
            return i
    return -1


# ====== 一致性哈希（扩容最少迁移） ======
class ConsistentHash:
    """一致性哈希 = 哈希环 + 虚拟节点"""

    def __init__(self, virtual_nodes: int = 150):
        self.ring = {}  # hash → node
        self.virtual_nodes = virtual_nodes
        self.sorted_hashes = []

    def add_node(self, node: str):
        for i in range(self.virtual_nodes):
            h = hashlib.md5(f"{node}:v{i}".encode()).hexdigest()
            hash_int = int(h, 16)
            self.ring[hash_int] = node
            self.sorted_hashes.append(hash_int)
        self.sorted_hashes.sort()

    def get_node(self, key: str) -> str:
        """顺时针找最近的节点"""
        h = int(hashlib.md5(key.encode()).hexdigest(), 16)
        # 二分查找
        import bisect
        idx = bisect.bisect_left(self.sorted_hashes, h)
        if idx == len(self.sorted_hashes):
            idx = 0  # 环形：回到起点
        return self.ring[self.sorted_hashes[idx]]


# ====== 分库分表后的路由 ======
class ShardingRouter:
    def __init__(self, db_count: int = 4, table_count: int = 16):
        self.db_count = db_count
        self.table_count = table_count

    def route(self, order_id: int):
        """根据订单ID路由到具体库和表"""
        db_idx = order_id % self.db_count       # 库：4个
        tbl_idx = order_id % self.table_count    # 表：16张
        return f"order_db_{db_idx}.order_{tbl_idx}"


# ====== 跨分片分页（面试重点） ======
def cross_shard_pagination(user_id: int, page: int, page_size: int):
    """
    跨分片分页 = 每个分片取前 N 条 → 归并排序 → 取第 page 页
    深分页性能差：翻到第100页要在每个分片取100*20=2000条，全量归并
    """
    all_shard_data = []
    for shard in range(4):
        # 每个分片取前 (page * page_size) 条
        data = query_shard(shard, f"SELECT * FROM orders WHERE user_id={user_id} "
                           f"ORDER BY create_time DESC LIMIT {page * page_size}")
        all_shard_data.extend(data)

    # 归并排序（按 create_time）
    all_shard_data.sort(key=lambda x: x['create_time'], reverse=True)

    # 取第 page 页
    start = (page - 1) * page_size
    return all_shard_data[start:start + page_size]

# 优化方案：禁止深分页 / 使用ES搜索 / 记录上次位置用游标分页
```

```python
        # 预热商品详情到本地缓存

    class RateLimiter:
        def __init__(self, qps: int = 1000):
            self.bucket = TokenBucket(rate=qps, capacity=qps * 2)

        def check(self) -> bool:
            return self.bucket.allow()

    LUA_DEDUCT_STOCK = """
    local stock_key = KEYS[1]
    local stock = tonumber(redis.call('get', stock_key) or 0)
    if stock > 0 then
        redis.call('decr', stock_key)
        return 1  -- 抢到了
    else
        return 0  -- 已售罄
    end
    """

        """尝试秒杀"""
        stock_key = f"seckill:stock:{self.product_id}"

        # 原子扣库存
        result = r.eval(self.LUA_DEDUCT_STOCK, 1, stock_key)
        if result == 0:
            return "已售罄"

        # 检查是否重复抢购（一人一单）
        if not r.set(f"seckill:user:{self.product_id}:{user_id}", "1", nx=True, ex=86400):
            # 回滚库存
            r.incr(stock_key)
            return "已参与过"

        # 抢到了！发MQ异步创建订单
        order_msg = {
            "product_id": self.product_id,
            "user_id": user_id,
            "timestamp": time.time()
        }
        mq_producer.send("seckill_order", json.dumps(order_msg))

        return "抢购成功，订单生成中..."

    # ====== 步骤3：MQ 消费者异步下单 ======
    def consumer_create_order(self, msg: dict):
        """消费秒杀消息，创建订单"""
        try:
            # 防超卖：DB 最终减库存
            affected = db.execute(
                "UPDATE product SET stock = stock - 1 "
                "WHERE id = ? AND stock >= 1",  # ← 关键：stock >= 1 防超卖

                print("库存不足，拒绝")
                return

                "INSERT INTO seckill_order (product_id, user_id, status) "
                "VALUES (?, ?, 'paid')",
                (msg['product_id'], msg['user_id'])
            )
            print("订单创建成功")
        except Exception as e:
            # 幂等处理：消息可能重复
            print(f"消费失败: {e}")
```

```python
"""
防雪崩五件套：熔断 + 降级 + 限流 + 隔离 + 超时
"""
import requests
import time
import threading
from concurrent.futures import ThreadPoolExecutor

class RobustServiceCaller:
    """健壮的服务调用：防御下游故障"""

    def __init__(self):
        # 1. 熔断器
        self.breaker = CircuitBreaker(failure_threshold=5, cooldown_seconds=10)

        # 2. 独立线程池（隔离）
        self.downstream_a_pool = ThreadPoolExecutor(max_workers=10, thread_name_prefix="svc-a")
        self.downstream_b_pool = ThreadPoolExecutor(max_workers=10, thread_name_prefix="svc-b")

        # 3. 限流
        self.limiter = TokenBucket(rate=100, capacity=200)

    def call_downstream(self, service: str, params: dict) -> dict:
        """调用下游服务，自带防护"""
        # 限流
        if not self.limiter.allow():
            return self.degrade(service, "限流")

        # 熔断
        try:
            result = self.breaker.call(
                self._do_call, service, params
            )
            return result
        except Exception:
            return self.degrade(service, "熔断/异常")

    def _do_call(self, service: str, params: dict) -> dict:
        """实际调用"""
        # 超时控制：连接超时1s，读取超时3s
        resp = requests.post(
            f"http://{service}/api/process",
            json=params,
            timeout=(1.0, 3.0)  # (connect_timeout, read_timeout)
        )
        resp.raise_for_status()
        return resp.json()

    def degrade(self, service: str, reason: str) -> dict:
        """降级：返回兜底数据"""
        if service == "inventory":
            return {"stock": 100, "source": "cache"}  # 返回缓存库存
        elif service == "user":
            return {"name": "匿名用户", "source": "default"}  # 返回默认值
        elif service == "recommend":
            return {"items": get_hot_products(10), "source": "hot"}  # 返回热门
```

```python
"""
分布式任务调度平台 Python 实现（简化版）
参考：PowerJob / Temporal 的架构
"""
import time
import json
import threading
import redis
import sqlite3
from enum import Enum

r = redis.Redis()

class TaskStatus(Enum):
    PENDING = "pending"
    RUNNING = "running"
    SUCCESS = "success"
    FAILED = "failed"

class DistributedScheduler:
    """
    调度中心（多副本HA） + Worker（执行器）
    """

    def __init__(self):
        self.db = sqlite3.connect('scheduler.db')
        self._init_db()

    def _init_db(self):
        self.db.execute("""
            CREATE TABLE IF NOT EXISTS job_definition (
                id INTEGER PRIMARY KEY,
                name TEXT,
                cron_expr TEXT,
                handler TEXT,       -- 执行的函数名
                params TEXT,        -- JSON 参数
                status TEXT,
                next_fire_time REAL
            )
        """)
        self.db.execute("""
            CREATE TABLE IF NOT EXISTS job_instance (
                id INTEGER PRIMARY KEY,
                job_id INTEGER,
                worker_id TEXT,
                status TEXT,
                start_time REAL,
                end_time REAL,
                result TEXT
            )
        """)

    # ====== 调度中心：时间轮 + DB持久化 ======
    def schedule_loop(self):
        """主循环：每秒扫描到期的任务"""
        while True:
            now = time.time()
            jobs = self.db.execute(
                "SELECT * FROM job_definition WHERE status='enabled' AND next_fire_time <= ?",
                (now,)
            ).fetchall()

            for job in jobs:
                # 选一个 Worker 分发（负载均衡：最少任务数）
                worker = self._select_worker()
                if worker:
                    self._dispatch(job, worker)
                    # 更新下次执行时间（cron计算）
                    next_time = self._cron_next(job['cron_expr'], now)
                    self.db.execute(
                        "UPDATE job_definition SET next_fire_time=? WHERE id=?",
                        (next_time, job['id'])
                    )

            self.db.commit()
            time.sleep(1)

    def _select_worker(self) -> str:
        """负载均衡：选任务最少的 Worker"""
        # 从注册中心获取在线 Worker 列表
        workers = self._get_online_workers()
        if not workers:
            return None
        # 选任务数最少的
        return min(workers, key=lambda w: w['task_count'])

    def _dispatch(self, job, worker):
        """分发任务到 Worker（通过 Redis/MQ）"""
        task_msg = {
            "job_id": job['id'],
            "handler": job['handler'],
            "params": json.loads(job['params']),
            "instance_id": f"{job['id']}_{int(time.time()*1000)}"
        }
        # 通过 Redis Stream 分发
        r.xadd(f"worker:{worker['id']}:tasks", task_msg)


class Worker:
    """
    执行器：注册发现 + 心跳 + 执行任务 + 上报结果
    """

    def __init__(self, worker_id: str):
        self.worker_id = worker_id

    def register(self):
        """启动时注册到注册中心"""
        r.hset("workers", self.worker_id, json.dumps({
            "id": self.worker_id,
            "ip": "10.0.0.1",
            "status": "online",
            "last_heartbeat": time.time()
        }))

    def heartbeat(self):
        """定时心跳（每5秒）"""
        def beat():
            while True:
                r.hset("workers", self.worker_id, json.dumps({
                    "id": self.worker_id,
                    "status": "online",
                    "last_heartbeat": time.time()
                }))
                time.sleep(5)
        threading.Thread(target=beat, daemon=True).start()

    def consume_tasks(self):
        """消费调度中心分发的任务"""
        while True:
            tasks = r.xreadgroup(
                "worker_group", self.worker_id,
                {f"worker:{self.worker_id}:tasks": ">"},
                count=1, block=5000
            )
            for stream, msgs in tasks:
                for msg_id, fields in msgs:
                    self._execute(fields['handler'], json.loads(fields['params']))
                    r.xack(f"worker:{self.worker_id}:tasks", "worker_group", msg_id)

    def _execute(self, handler: str, params: dict):
        """执行任务 + 指数退避重试"""
        max_retries = 3
        for attempt in range(max_retries):
            try:
                # 动态调用处理函数
                func = getattr(task_handlers, handler)
                result = func(**params)
                self._report_result(params['instance_id'], "success", result)
                return
            except Exception as e:
                if attempt < max_retries - 1:
                    # 指数退避：1s → 5s → 25s
                    wait = 5 ** attempt
                    print(f"重试 {attempt+1}/{max_retries}，等待{wait}s")
                    time.sleep(wait)
                else:
                    self._report_result(params['instance_id'], "failed", str(e))
```

# 九、知识自查清单 — 速记答案

## 必答对（答错大概率挂）

| 考点 | 一句话答案 | Python关键词 |
|------|-----------|-------------|
| CAP定理 | P必须选，实际CP vs AP：ZK(CP)选一致，Eureka(AP)选可用 | `if network_partition: choose CP or AP` |
| 消息不丢失 | 生产confirm + Broker刷盘 + 消费手动ACK | `auto_ack=False`, `delivery_mode=2` |
| 消息重复消费 | MQ保证At-Least-Once，幂等靠唯一索引/SETNX/版本号 | `r.set(key, nx=True)`, `INSERT ... ON DUPLICATE KEY` |
| 缓存穿透 | 查不存在的数据 → 布隆过滤器 + 缓存空值 | `BloomFilter().might_contain(key)` |
| 缓存击穿 | 热点过期 → 互斥锁单线程查DB | `r.set(lock_key, nx=True)` |
| 缓存雪崩 | 大批同时过期 → 过期时间加随机值 | `ttl = base + random(0,600)` |
| 分布式事务 | RocketMQ事务消息：半消息→本地事务→Commit/Rollback | `send_half_msg() → do_local_tx() → commit/rollback()` |
| 分布式锁 | Redisson看门狗：加锁启动续期线程，每10s续期 | `watchdog.renew_every(ttl/3)` |
| MySQL最左前缀 | (a,b,c)索引 → a能走，a+b能走，b单独不行 | `WHERE a=1 AND b=2` ✓, `WHERE b=2` ✗ |
| MVCC | ReadView + Undo Log版本链 → 读不阻塞写 | `trx_id` + `roll_pointer` 多版本 |
